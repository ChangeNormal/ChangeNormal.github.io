def binary_search(list, item):
    low = 0
    high = len(list)-1
    while low <= high:
        mid = (low + high) // 2 # Ëøô‰∏™Âú∞ÊñπÊ≥®ÊÑèPython3.x‰∏éPython2.xÊúâÂ∑ÆÂºÇ
        guess = list[mid]
        if guess == item:
            return mid
        if guess < item:
            low = mid + 1
        else:
            high = mid -1
    return None
